#!/bin/bash ESC=$(printf '\033') TARGET_DIR=/home/$USER/.config/systemd/user function helper_gen() { cat < "${TARGET_DIR}/code-tmp" exit 0 EOS } function service_gen() { cat </dev/null; then printf "${ESC}[31m%s${ESC}[m\n" "✖ $cmd is not installed. Please install it and try again." exit 1 fi done if [ "$(ps -p 1 -o comm=)" != "systemd" ]; then printf "${ESC}[31m%s${ESC}[m\n" "✖ This script is only for systemd" exit 1 fi if [ "$(loginctl show-user "$USER" | grep 'Linger=yes')" != "Linger=yes" ]; then enable_linger fi if [ "$(systemctl --user is-active code.service)" == "active" ]; then printf "${ESC}[33m%s${ESC}[m\n" "✅ code.service is already running" exit 0 fi mkdir -p "${TARGET_DIR}" echo -e "$(helper_gen)" >"${TARGET_DIR}/code-helper" echo -e "$(service_gen)" >"${TARGET_DIR}/code.service" systemctl --user daemon-reload systemctl --user enable code.service systemctl --user start code.service exit 0
